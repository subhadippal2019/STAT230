\begin{frame}\frametitle{Absolute Value}
Absolute value of a real number is the magnitude of the real number ignoring its sign.  Formally, we have the following definition. 
\define{Absolute Value}{
Let $\HLTY{x}\in \R$ be any real number, then the {\bf absolute value of} $\HLTY{x}$ (denotes as $\abs{x}$) is defined to be 
$$\HLTW{ \abs{\HLTY{x}}= 
\begin{cases}
\HLTY{x} & \text{ if} \HLTY{x}\geq 0, \\
-\HLTY{x} & \text{ if }  \HLTY{x}< 0, \\
\end{cases}}$$
}


\begin{itemize}
\item $\abs{5}= $ 
\item $\abs{-7.6}=  $
\item $\abs{0}=  $
\item $\abs{1005.7}=  $
\item $\abs{-200}= $
\end{itemize}


\end{frame}



\begin{frame}
\define{Indicator Function}{
Let $A$ be a set.  The  {\bf  indicator function for the set $A$}, denoted by $\Indicator{A}{x}$, is defined to be 
$$ \HLTW{ \Indicator{A}{x}:=
 \begin{cases}
 1 & \text{if } x\in A\\
  0 & \text{if } x\notin A.
\end{cases}  }$$
}

\end{itemize}
\end{frame}



\begin{frame}\frametitle{A Few Mathemetical Function and Notation}

\define{Factorial}{
Let $n$ be a {\bf non-negative integer,} then the {\bf factorial of n}, denoted as $\HLTY{n!}$ is defined to be
\begin{enumerate}
\item $0!=\HLTY{1}$, 
\item $1!=\HLTY{1}$, 
\item $n!= \HLTY{n\times(n-1)\times \ldots \times 1}$ for $n\geq 2$
\end{enumerate} 
}

\begin{frame}\frametitle{n choose r, ${n \choose r}$}
\define{}{
Let $n, r$ be two {\bf non-negative integes,} such that $r\leq n$,  then the {\bf n choose r}, denoted by $\HLTY{{n\choose r}}$, is defined to be $${n \choose r}:= \frac{n!}{\left(r!\right) \times \left((n-r)!\right)}$$
\vspace{-.1in}
}

\qBrd[4.6in]{teal!40}{
\qBrd[.5in]{teal!70}{Result:} \tiny If $n, r$ be two {\bf non-negative integes,} such that $r\leq n$,  then $\HLTY{{n\choose r}}= \HLTY{{n\choose (n-r)}} $
}

\begin{itemize}
\item $\HLTY{{ 0\choose 0}= 1}$,  $\HLTY{{ 1\choose 0}= 1}$, $\HLTY{{ 5\choose 3}=\frac{5!}{3!\times ((5-3)!)}=\frac{5!}{3!\times (2!)}= \frac{5\times 4 \times 3}{3\times 2\times 1}= 10. }$
\item ${ 10\choose 2}=$
\item ${ 100\choose 95}=$
\end{itemize}


\end{frame}

\begin{frame}\frametitle{The mathemetical constant $e$}
\define{"e"}{
The mathemetical constant $\HLTY{e}$ is an transcendental real number given by, 
$$\HLTY{e}=\HLTW{\displaystyle \sum_{n=0}^{\infty}\frac{1}{n!}} = \HLTY{\displaystyle 1+\frac{1}{1!}+\frac{1}{2!}+ \frac{1}{3!}+\ldots,}  $$
Approxiimately $e\approx  2.7183$
}



\end{frame}

\begin{frame}\frametitle{Exponential Series $e^x \text{ or }( \exp(x))$}
\vspace{-.1in}
\define{Exponential Series}{
For any real number $x\in \R$, the exponential series $\HLTY{e^x}$  (or sometimes denoted as $\HLTY{\exp(x)}$) is defined as, 
$$\HLTY{\displaystyle e^{x}}=\HLTW{\displaystyle \sum_{n=0}^{\infty}\frac{x^n}{n!}} = \HLTY{\displaystyle 1+\frac{x}{1!}+\frac{x^2}{2!}+ \frac{x^3}{3!}+\cdots,}  $$
}

}
