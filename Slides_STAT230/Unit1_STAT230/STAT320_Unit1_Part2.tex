Let $\HLTY{x}\in \R$ be any real number, then the {\bf absolute value of} $\HLTY{x}$ (denotes as $\abs{x}$) is defined to be 
$$\HLTW{ \abs{\HLTY{x}}= 
\begin{cases}
\HLTY{x} & \text{ if} \HLTY{x}\geq 0, \\
-\HLTY{x} & \text{ if }  \HLTY{x}< 0, \\
\end{cases}}$$
}

\begin{frame}\frametitle{n choose r, ${n \choose r}$}
\define{}{
Let $n, r$ be two {\bf non-negative integes,} such that $r\leq n$,  then the {\bf n choose r}, denoted by $\HLTY{{n\choose r}}$, is defined to be $${n \choose r}:= \frac{n!}{\left(r!\right) \times \left((n-r)!\right)}$$
\vspace{-.1in}
}

\qBrd[4.6in]{teal!40}{
\qBrd[.5in]{teal!70}{Result:} \tiny If $n, r$ be two {\bf non-negative integes,} such that $r\leq n$,  then $\HLTY{{n\choose r}}= \HLTY{{n\choose (n-r)}} $
}

\begin{itemize}
\item $\HLTY{{ 0\choose 0}= 1}$,  $\HLTY{{ 1\choose 0}= 1}$, $\HLTY{{ 5\choose 3}=\frac{5!}{3!\times ((5-3)!)}=\frac{5!}{3!\times (2!)}= \frac{5\times 4 \times 3}{3\times 2\times 1}= 10. }$
\item ${ 10\choose 2}=$

\end{itemize}


\end{frame}
