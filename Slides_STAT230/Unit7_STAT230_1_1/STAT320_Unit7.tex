\section{Discrete Multivariate Random Variables }

We will just restrict the presentation to the bivariate case.

\begin{frame}\frametitle{Joint Cumulative Distribution Function}

\define{Bivariate CDF}{
Let X, Y be two discrete random variables. The joint cumulative distribution function is given by
$$F_{_{X,Y}}(\HLTY{x},\HLTY{y}):= P\left(X\leq \HLTY{x}, Y\leq \HLTY{y} \right) .$$
}

\qBrd{olive!40}{
If  the joint probability mass function of X and Y is  $\pmf_{_{X,Y}}(x, y) = P(X = x; Y = y).$ then 
$$ P\left(X\leq \HLTY{x}, Y\leq \HLTY{y} \right)= \mathop{\sum\sum}_{\left\{ \HLTW{s}\leq \HLTY{x}, \HLTW{t} \leq \HLTY{y} \\\text{ where } (s,t)\in \HLTW{\support[ X,Y]}\right\}} \pmf_{_{X,Y}}(\HLTW{s}, t) $$
}
\end{frame}


\begin{frame}\frametitle{Marginal  Distributions}
\qbx[4.2in]{babyblue!40}{
The marginal probability mass function of $X$ is given by
$$\pmf_{_X}(\HLTY{x})=  \sum_{(x,\HLTW{t})\in \support[XY]}  \pmf_{_{X,Y}}(\HLTY{x}, \HLTW{t})$$
}
\vspace{.1in}

\qbx[4.2in]{babyblueeyes!40}{
The marginal probability mass function of $X$ is given by
$$\pmf_{_Y}(\HLTY{y})=  \sum_{(\HLTW{s},\HLTY{y})\in \support[XY]}  \pmf_{_{X,Y}}(\HLTW{s}, \HLTY{y})$$
}

\end{frame}

\section{Conditional Distributions}
\TransitionFrame[bittersweet]{\Large Conditional Distributions  }



\begin{frame}
\define{}{
If $\pmf_{_{XY}}(x, y)$ denotes the joint probability function of two discrete random variables X and Y and if $\pmf_X(x)$ and $\pmf_Y (y)$ denote the marginal probability function of $X,$ ($Y $ respectively) then the conditional probability of X given $Y = y $ is given by $$\pmf_{_{X\mid Y}}(x\mid y)=  \frac{\pmf_{_{X,Y}}(x,y)}{\pmf_{_Y}(y)}$$
The conditional probability of Y given X = x is given by
$$\pmf_{_{ Y\mid X}}(y \mid x)=  \frac{\pmf_{_{X,Y}}(x,y)}{\pmf_{_X}(x)}.$$
}
\end{frame}

\begin{enumerate}[a).]
\item Find the conditional probability of X given Y = 2. 
\item  Use this to compute $P(X\leq 2\mid Y = 2).$
\end{enumerate}

\define{Independent Random Variables}{
Two random variables X and Y are said to be independent if $ f_{_{X,Y}}(x, y) = f_{_X}(x) f_{_Y} (y)$ for all x and y's,
}
\end{frame}

0 &  \text{otherwise}\end{cases}$$
\begin{enumerate}[a).]
\item Are X and Y independent?
\end{enumerate}
}\\

}

\section{Expectation for  Different Functions of Multivariate  Random Variables}
\TransitionFrame[bittersweet]{\Large Expectation for  Different Functions of Multivariate  Random Variables}


\begin{frame}
\qbx[4.2in]{babyblue!40}{
Let X, Y be two discrete random variables with joint probability function $\pmf_{_{X,Y}}(x, y)$. Then the expected value of $g(X, Y )$ is given by
$$E\left( g(X,Y) \right)=  \sum_{(x,y)\in \support[XY]} g(x,y) \pmf_{_{X,Y}}(x,y)$$
}
\vspace{.1in}

\qbx[4.2in]{babyblueeyes!40}{
Let X, Y be two continuous random variables with joint probability density function $f_{_{X,Y}}(x, y)$. Then the expected value of $g(X, Y )$ is given by
$$E\left( g(X,Y) \right)=  \int\limits_{(x,y)\in \support[XY]} g(x,y) f_{_{X,Y}}(x,y) dx\; dy$$
}
\end{frame}

\section{Variance and Covariance of a Random Variable }
\begin{frame}\frametitle{Reminder: Mean and Variance of a Random Variable}
\qbx[4.6in ]{amethyst!60}{
\qBrd[4.5in ]{amber!40}{\qBrd[.8in ]{amber!60}{Mean:} Let $X$ be a random variable, then  $E(X)$ denoted by $\mu_{_X}$ is called the {\bf mean} of the random variable. }\\
\qBrd[4.5in ]{olive!40}{ \qBrd[.8in ]{olive!60}{Variance:} Let $X$ be a random variable, then  $E\left(X-\mu_{_X}\right)^2$ denoted by Var${(X)}$ is called the {\bf Variance} of the random variable.  Note that, the alternative formula for variance is: 
 $$ \text{Var}(X):= E(X^2)- \left(E(X)\right)^2.$$
}
}


\end{frame}


\begin{frame}\frametitle{Covariance}
\begin{center}
\define{Covariance}{
Let $X$, and $Y$ be two random variables with a joint distribution.  Then $$\text{Cov}(X,Y)=E \left( (X-\mu_{_X}) (Y-\mu_{_Y})\right), $$
where $\mu_{_X}$ and $\mu_{_Y}$ denotes the mean of the random variables $X$, and $Y$ respectively. \\
\vspace{.1in}
}
\vspace{-.15in}
\qBrd[4.2in ]{amethyst!30}{
An Alternative Formulation for the covariance is the following:\\
$$\qBrd[2.2in ]{amethyst!50}{{\text{Cov}(X,Y)= E(XY)- E(X)E(Y) }}$$
}
\end{center}

\end{frame}

\begin{frame}\frametitle{Statistically Independent Random Variables and Covariance}

\begin{theorem}
If $X$, and $Y$ are two {\bf statistically independent} random variables, then 
$$\text{Cov}(X,Y)=0 $$.
However, the converse of the result is not true in general. 
\end{theorem}
\end{frame}

\TransitionFrame[antiquefuchsia]{\Large Expected Value of Linear Combination  }

\begin{frame}


\begin{center}
\qbx[4.6in ]{amethyst!40}{
Let $X_1, X_2, \ldots X_n$ are random variables and $Y =a_0+ \sum_{i=1}^{n} a_i X_i $, where $a_i's$ are constants then 
\begin{enumerate}
\item \qBrd[4.1in ]{amber!40}{$E(Y)=  a_0+\sum_{i=1}^{n} a_i E\left(X_i\right) $}
\item \qBrd[4.1in ]{applegreen!40}{$\text{Var}(Y)=  \sum_{i=1}^{n} a^2_i \text{Var}\left(X_i\right)+ 2{\sum\sum}_{1\leq i <j \leq n}  a_i a_j \text{Cov}(X_i, X_j) $}
\end{enumerate}
}

 \qBrd[4.1in ]{green!40}{If $X_1, X_2, \ldots X_n$are 
 mutually statistically independent then, 
 $$\text{Var}(Y)=  \sum_{i=1}^{n} a^2_i \text{Var}\left(X_i\right)$$}
 \end{center}
\end{frame}



\begin{frame}


\begin{center}
\qbx[4.6in ]{amethyst!40}{
Let $X_1, X_2, \ldots X_n$are random variables and $Y_1 =a_0+ \sum_{i=1}^{n} a_i X_i $, and $Y_2 =b_0+ \sum_{i=1}^{n} b_i X_i $, where $a_i's$ abd $b_i's$ are constants then 
\qBrd[4.2in ]{applegreen!40}{$\text{Cov}(Y_1, Y_2)=  \sum_{i=1}^{n} a_i b_i \text{Var}\left(X_i\right)+ 2{\sum\sum}_{1\leq i <j \leq n}  a_i b_j \text{Cov}(X_i, X_j) $}
}

 \qBrd[4.1in ]{green!40}{If $X_1, X_2, \ldots X_n$are 
 mutually statistically independent then, 
 $$\text{Cov}(Y)=  \sum_{i=1}^{n} a_i b_j \text{Var}\left(X_i\right)$$}
 \end{center}
\end{frame}
