\DefBoxOne{}{\color{black}
A few Functions: Here $0<\pi<1$ and $x\in \R$.
\begin{center}
\begin{tabular}{ |c|| c|}
\hline
$\text{Odds}$ & $\displaystyle \text{Odds}(\pi) = {\frac{\pi}{1-\pi}}$ \\
\hline
$\text{Logit}$ & $\displaystyle \text{Logit}(\pi) = {\log\left(\frac{\pi}{1-\pi}\right)}$ \\
\hline
$\text{Expit}$ & $\displaystyle \text{Expit}(x) = \frac{\exp(x)}{1+\exp(x)}$ \\
\hline
$\text{ReLU}$ & $\displaystyle \text{ReLU}(x) = \max(x,0)$ \\
\hline
$\text{tanh}$ & $\displaystyle \text{tanh}(x) =\frac{\exp(x)-\exp(-x)}{\exp(x)+\exp(-x)}$ \\
\hline
$\text{Sigmoid}$ & $\displaystyle \text{Sigmoid}(x) =\frac{1}{1+\exp(-x)}$ \\
\hline
$\text{PReLU}$ & $\displaystyle \text{PReLU}(x) =\max(\alpha x, x)$ for $0<\alpha<1$ \\
\hline
\end{tabular}
\end{center}
